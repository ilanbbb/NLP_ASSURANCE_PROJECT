# pip install lime
from lime.lime_text import LimeTextExplainer

explainer = LimeTextExplainer(class_names=le.classes_)

def predict_proba_pipeline(texts):
    """Pipeline pour LIME : texte brut → probabilités."""
    X = tfidf.transform(texts)
    return clf.predict_proba(X)

# Topic Modeling avec LDA
# pip install pyLDAvis
from sklearn.decomposition import LatentDirichletAllocation
from sklearn.feature_extraction.text import CountVectorizer
import pyLDAvis
import pyLDAvis.lda_model

# Vectorisation pour LDA (CountVectorizer, pas TF-IDF)
count_vec = CountVectorizer(
    max_features=5000,
    min_df=5,
    max_df=0.95,
    ngram_range=(1, 1)
)
X_counts = count_vec.fit_transform(df_train['avis_clean'].dropna())

# Entraînement LDA
N_TOPICS = 6  # Nombre de topics à découvrir
lda = LatentDirichletAllocation(
    n_components=N_TOPICS,
    random_state=42,
    max_iter=20,
    learning_method='online'
)
lda.fit(X_counts)
print(f'✅ LDA entraîné avec {N_TOPICS} topics')

# Affichage des topics
feature_names = count_vec.get_feature_names_out()

# Noms des topics (à interpréter manuellement après avoir vu les mots)
topic_names = [
    'Service Client',
    'Sinistre & Remboursement',
    'Prix & Tarifs',
    'Résiliation & Contrat',
    'Satisfaction Générale',
    'Délais & Réactivité'
]

fig, axes = plt.subplots(2, 3, figsize=(16, 10))
axes = axes.flatten()

for topic_idx, (ax, topic_name) in enumerate(zip(axes, topic_names)):
    top_word_idx = lda.components_[topic_idx].argsort()[-12:][::-1]
    top_words = [feature_names[i] for i in top_word_idx]
    top_weights = lda.components_[topic_idx][top_word_idx]
    top_weights = top_weights / top_weights.sum()  # normaliser

    ax.barh(top_words[::-1], top_weights[::-1],
            color=sns.color_palette('tab10')[topic_idx])
    ax.set_title(f'Topic {topic_idx+1} : {topic_name}', fontweight='bold', fontsize=10)
    ax.set_xlabel('Poids')

plt.suptitle('Topics LDA — Thèmes détectés dans les avis clients', fontsize=13, fontweight='bold')
